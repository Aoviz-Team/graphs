# 水印插件

## 使用方式

通过传入文本内容渲染水印


```jsx live=true template=configPlayground
import { Force, WatermarkPlugin, DEFAULT_LAYOUT_OPTION } from '@aoviz/graphs'
import { ONE_DEGREE_BANK_DATA } from '@aoviz/mock'
import { useRef, useEffect } from 'react'
import { useControls } from 'leva'

const option = {
  link:{
    label: {
      backgroundColor: 'rgba(114, 193, 209, 0.6)' 
    }
  },
  layout: DEFAULT_LAYOUT_OPTION
}
export default function Viz() {
  const wrapper = useRef(null);
  const instanceRef = useRef(null)
  const config = useControls({
    rotate: 30,
    size: 22,
    color: 'rgba(179, 179, 255, 0.5)',
    value: '@aoviz/graphs'
  })

  useEffect(() => {
    instanceRef.current = new Force(wrapper.current,{ data: ONE_DEGREE_BANK_DATA, option });
  }, [wrapper]);

  useEffect(() => {
    instanceRef.current.setPlugins([new WatermarkPlugin(config)])
  }, [instanceRef, config])

  return (
    <div style={{ height: 380 }}>
		  <div ref={wrapper} style={{ height: 380 }}></div>
	  </div>
  )
}

```
